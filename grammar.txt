<program> ::= <item>*

<item> ::= <attribute>* ( <func> | <extern_func> )

<func> ::= <func_sig> <block>
<extern_func> ::= extern <func_sig> ;
<func_sig> ::= func <ident> \( <param_list> \) <type>?
<param_list> ::= <param> | <param> , <param_list> | <epsilon>
<param> ::= <ident> <type>

<type> ::= <primitive>
<primitive> ::= int | bool | \(\)

<stmt> ::= <expr_stmt> | <let>
<expr_stmt> ::= <f_expr> ; | <cf_expr> ;?
<let> ::= let <ident> <type>? = <expr> ;

<expr> ::= <f_expr> | <cf_expr>

<f_expr> ::= <constant> | <unary> | <binary> | <group> | <call> | <return>
<constant> ::= <integer> | <boolean> | \(\)
<unary> ::= <unary_op> <expr>
<unary_op> ::= ! | -
<binary> ::= <expr> <binary_op> <expr>
<binary_op> ::= + | - | && | \|\| | = | == | != | < | <= | > | >=
<group> ::= \( <expr> \)
<call> ::= <expr> \( <arg_list> \)
<arg_list> ::= <expr> | <expr> , <arg_list> | <epsilon>
<return> ::= return <expr>?

<cf_expr> ::= <block> | <if> | <loop>
<block> ::= { <stmt>* <expr>? }
<if> ::= if <expr> <block> <else>?
<else> ::= else <block>
<loop> ::= loop <block>
